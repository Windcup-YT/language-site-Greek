function checkAnswer(selected) { speak(currentQuestion.word); // 讓語音讀出希臘文單字 const feedback = document.getElementById("feedback"); if (selected === currentQuestion.translation) { feedback.textContent = "Correct! 🎉"; feedback.style.color = "lightgreen"; } else { feedback.textContent = `Oops! The correct answer is: ${currentQuestion.translation}`; feedback.style.color = "red"; } setTimeout(() => { nextQuestion(); }, 1500); } function speak(text) { speechSynthesis.cancel(); // 先停止其他語音，避免疊音 const utterance = new SpeechSynthesisUtterance(text); utterance.lang = "el-GR"; // 希臘語 utterance.rate = 0.9; speechSynthesis.speak(utterance); }
